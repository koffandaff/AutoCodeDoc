Example Configuration

from dataclasses import dataclass, field
from typing import Optional


@dataclass
class ModelConfig:
    """
    Configuration for the GNN model architecture.

    Attributes
    ----------
    hidden_dim : int
        Hidden dimension size for all MLPs. Controls model capacity.
    hidden_layers : int
        Number of hidden layers in each MLP block.
    processor_layers : int
        Number of message-passing steps in the mesh processor.
    mesh_aggr : str
        Aggregation method for mesh GNN. One of ``"sum"`` or ``"mean"``.
    output_std : bool
        Whether to predict uncertainty (standard deviation) alongside mean.
    model_type : str
        Model variant to use. One of ``"graph_lam"``, ``"hi_lam"``,
        ``"hi_lam_parallel"``.
    """

    hidden_dim: int = 64
    hidden_layers: int = 1
    processor_layers: int = 4
    mesh_aggr: str = "sum"
    output_std: bool = False
    model_type: str = "hi_lam"


@dataclass
class OptimizerConfig:
    """
    Configuration for the optimizer and learning rate schedule.

    Attributes
    ----------
    lr : float
        Initial learning rate.
    weight_decay : float
        L2 regularization coefficient.
    scheduler : str
        Learning rate scheduler type. One of ``"cosine"``, ``"step"``,
        ``"plateau"``.
    warmup_steps : int
        Number of linear warmup steps before applying the scheduler.
    max_epochs : int
        Maximum number of training epochs.
    patience : int
        Early stopping patience (epochs without improvement).
    """

    lr: float = 1e-3
    weight_decay: float = 0.0
    scheduler: str = "cosine"
    warmup_steps: int = 1000
    max_epochs: int = 200
    patience: int = 30
